기본 화면구성 

<!------------------------------------------------------->
<!-- 파일설명 주석내용 위치 -->
<!------------------------------------------------------->
<%@ page language="java" contentType="text/html; charset=EUC-KR" pageEncoding="EUC-KR"%>
<%@ page errorPage="../aviscmn/error.jsp" %>
<%@ include file="../aviscmn/common.jsp" %>
<!------------------------------------------------------->
<!-- JSP Page Import 설정 위치 -->
<!-- Java 변수 및 데이터 초기설정 위치 -->
<!------------------------------------------------------->
<!DOCTYPE html>
<html>
<head>
<meta http-equiv="Content-Type" content="text/html; charset=EUC-KR">
<title>타이틀명</title>
<!-- 작업하는 파일위치에 따른 include common_script 파일경로수정  -->
<%@ include file="./resources/common/common_script.jsp" %>
<!------------------------------------------------------->
<!-- 화면에서 사용할 js 파일설정 위치 -->
<!------------------------------------------------------->
</head>
    <body>
        <div id="wrap">
            <!-- 팝업화면 타이틀 시작 -->
            <div class="header_pop">
                <img src="./resources/images/ui/bg_poptopl.gif" width="28" height="36">
                <label id="txtTitle">팝업타이틀</label>
            </div>
            <!-- 팝업화면 타이틀 종료 -->
            <!-- 콘텐츠 화면내용 시작 -->
            <div id="content_table">
            </div>
            <!-- 콘텐츠 화면내용 종료 -->
        </div>
        <!-- JavaScript 시작 -->
        <script type="text/javascript">
        </script>
        <!-- JavaScript 종료 -->
    </body>
</html>
※ 참조
-   주석으로 표시된 위치해 각 해당하는 설정 및 내용을 삽입한다.

콘텐츠 입력 기본구성 

<!-- 콘텐츠 화면내용 시작 -->
<div id="content_table">
    <!-- 입력폼 시작 -->
    <div class="tb_basic">
        <table>
            <colgroup>
                <col style="width:20%;">
                <col style="width:30%;">
            </colgroup>
            
            <tbody>
                <tr>
                    <th></th> <!-- 제목 -->
                    <td></td> <!-- 입력값 -->
                </tr>
            </tbody>
        </table>
    </div>
    <!-- 입력폼 종료 -->
    
    <!-- 버튼영역 시작 -->
    <div class="btn_box">
        <ul>
            <li>
                <span class="button black large">
                    <input type="button" onclick="" value="" />
                </span>
            </li>
        </ul>
    </div>
    <!-- 버튼영역 종료 -->
</div>
<!-- 콘텐츠 화면내용 종료 -->
※ 참조
<colgroup>
-   <col>태그를 그룹으로 관리하며, 열 단위로 스타일 제어, <col>을 포함하는 부모요소

<ul>
-   버튼 목록의 행

<li>
-   버튼 목록의 열

    <div class="btn_box">
        <ul>
            <li>
                <span class="button black large">
                    <input type="button" onclick="" value="" />
                </span>
            </li>
        </ul>
    </div>
    <!-- 버튼영역 종료 -->
    
    <!-- 그리드영역 시작 -->

</div>
