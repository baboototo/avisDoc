TO-BE

※ 중요사항

• Type:"Status"
   -   Sheet 컬럼 값을 수정 또는 입력할 경우 자동으로 상태 값이 변경

• SaveName:"DATA_KEY"
   -   맵핑된 DataSet Row Index
   -   조회된 Sheet 행을 오름차순, 내림차순 했을 경우 보여지는
       Sheet Row Index가 아닌 DataSet의 Row Index

var headers = [{Text: "상태|DATA_KEY|순번|고객코드|사업자명|사업자번호", Align:"Center"}];

-  조회 기능이 있는 화면에서 사용 (Sheet가 없는 등록화면에서는 미사용)

-  Event 상세 설명 : IBSheet 개발가이드 문서 참고

Grid Function

• Row Index 반환

   /* AS-IS */
   var rowPos = dataSet01.RowPosition;

   /* TO-BE */
   var rowPos = sheet.GetSelectRow();

• Grid 전체 데이터 삭제

   /* AS-IS */
   dataSet01.Cleardata();

   /* TO-BE */
   sheet.RemoveAll();   // 헤더 행을 제외한 모든 데이터 행을 지운다.

   sheet.Reset();           // IBSheet에 설정된 모든 기본 속성을 제거하고 초기상태로 변경한다.

• Grid 데이터 입력/수정여부 확인

   /* AS-IS */
   dataSet01.IsUpdated

   /* TO-BE */
   sheet.IsDataModified())

• Grid 데이터 입력/수정여부 확인

   /* AS-IS */
   if (trSet01.ErrorCode == 0) {
       alert("<%=Constants.INSERT_CONFIRM %>");
       dataSet02.SyncLoad = true;
       dataSet01.ClearData();
       dataSet01.ImportData(dataSet02.ExportData(1, dataSet02.CountRow, true));
       dataSet01.ResetStatus();
   }

   /* TO-BE */
   if (trSet01.ErrorCode == 0) {
       alert("<%=Constants.INSERT_CONFIRM %>");
       LoadSearchDataFromDataSet(sheet, dataSet02);
   }

• Data ExportData

   /* AS-IS */
   dataSet02.ExportData(1, dataSet02.CountRow, true);

   /* TO-BE */
   dataSet02.ExportData(StartRow, EndRow);

• Grid 행 추가

   /* AS-IS */
   dataSet01.AddRow();

   /* TO-BE */
   sheet01.DataInsert(0);   // 첫 행 추가
   sheet01.DataInsert(-1);  // 마지막 행 추가

• Grid 행 삭제

   /* AS-IS */

   /* TO-BE */
   sheet01.RowDelete(Row);   // 행 삭제
   sheet01.SetCellValue(Row, "ISTATUS", "D");  // 상태 값 "D" 적용 (삭제 상태 값)

</script>

/** Sheet명_OnClick() */
function sheet_OnClick(Row, Col, Value){
    /** 선택한 Row 값이 헤더행의 개수 보다 작으면 return */
    if(Row < sheet.HeaderRows()) return;
    
    /** 선택한 Column의 SaveName */
    var colid = sheet.ColSaveName(Col);
}

 OnUserColor Event

※ Gauce Grid Format에서 사용되는 OnUserColor 이벤트는 화면로딩시
    Row, Column 상태값에 따라 Row Color, Column Color, Font Color 설정 하는 이벤트가 별도로 존재한다.
    하지만 IBSheet Grid 에는 이벤트 함수가 별도로 존재 하지 않는다. Gauce Grid 발생하는 이벤트 코드는
    OnLoadData 함수 안에서 이벤트를 구현해야 한다.

AS-IS 
<script language="JavaScript" for="grdSheet01" event="OnUserColor(row,eventid)">
    if(eventid == "ucolor") {
        var gubun = jsNvl(dataSet01.NameValue(row, 'USEYN'));

        if ( gubun == "Y") {
            this.ColumnProp('CUSTNAME', 'Edit') = "none";
            return "#E6E6E6";
        } else {
            this.ColumnProp('CUSTNAME', 'Edit') = "";
            return "white";
        }
    }
</script>
TO-BE 
/** Sheet명_OnLoadData() */
function sheet_OnLoadData(data){
    var jsonData = JSON.parse(data);
		
    if(jsonData.hasOwnProperty("Data")){
        var rowDatas = jsonData.Data;
		
        $.each(rowDatas, function(){
            // 조회된 JSON 데이터 수정 및 속성값 설정
            
            /** USEYN의 값 */
            var gubun = this.USEYN;
            
            if(gubun == "Y"){
                
                /** 요소 변경 Sample Start */
                this["CUSTNAME#Edit"] = false;               // CUSTNAME 컬럼 값 수정 불가
                this["CUSTNAME#BackColor"] = "#E6E6E6";      // CUSTNAME 컬럼 배경 색상 변경 
                this["CUSTNAME#FontColor"] = "#E6E6E6";      // CUSTNAME 컬럼 글자 색상 변경
                this["BackColor"] = "#E6E6E6";               // 행 배경 색상 변경
                this["FontColor"] = "#E6E6E6";               // 행 글자 색상 변경
                /** 요소 변경 Sample End */
                
            }else{
                this["CUSTNAME#Edit"] = true;                // CUSTNAME 컬럼 값 수정 가능
                this["CUSTNAME#BackColor"] = "#FFFFFF";      // CUSTNAME 컬럼 배경 색상 변경 
            }
            
        });
    }
		
    return JSON.stringify(jsonData);
}
