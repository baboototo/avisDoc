Grid Script 선언

※ Gauce Grid Script 선언 방식을 IBSheet Script 선언으로 변경

AS-IS 
<script language="javascript" src="../aviscmn/js/grid.js"></script>
<script language="javascript" src="../aviscmn/js/dataset.js"></script>
<script language="javascript" src="../aviscmn/js/transset.js"></script>

<%@ include file="../aviscmn/inc/condSet.jsp"%>
<%@ include file="../aviscmn/inc/dataSet01.jsp"%>
<%@ include file="../aviscmn/inc/dataSet02.jsp"%>
<%@ include file="../aviscmn/inc/trSet01.jsp"%>
<%@ include file="../aviscmn/inc/bindCtrl01.jsp"%>
TO-BE 
<script type="text/javascript">
    var condSet   = new DATASET("condSet");
    var dataSet01 = new DATASET("dataSet01");
    var dataSet02 = new DATASET("dataSet02");
    var trSet01   = new TRSET("trSet01");
</script>
TO-BE (generateSql) 
<script type="text/javascript">
    var condSet   = new DATASET("condSet");
    var dataSet01 = new DATASET("dataSet01", true); // dataSet01이 generateSql 기능으로 조회 될 경우 true 적용
    var dataSet02 = new DATASET("dataSet02");
    var trSet01   = new TRSET("trSet01");
</script>

DataSet / TrSet 초기화 설정

※ 변경 전에 사용하던 초기화 하는 코드 삭제
- initDataSet / initTransSet / initGrid

AS-IS 
function initWindow() {
    setTitle();
    _setEventHandler();
    initPopup();

    initDataSet(dataSet01);
    initTransSet(trSet01);
    initGrid(grdSheet01, true);
}
TO-BE 
function initWindow() {
    setTitle();
    _setEventHandler();
    initPopup();
}

 Gauce DataSet -> Script DataSet Function

※ Gauce DataSet 에서 사용하던 Function 명은 동일하나 사용법은 변경
-  Setter Function (ColumnString, ColumnValue, NameString, NameValue 등..)
    DATASET Object에 값을 삽입할 경우 기본방식과 다르게 특정함수에 파라미터값을 넘기도록 변경

AS-IS 
grdSheet01.DataID = "dataSet01";
grdSheet01.Format = '' + '<C>ID=Currow   Name="순번"     width=30    align=center    Value={String(CurRow)}    </C>'
                       + '<C>ID=Code     Name="고객코드"  width=70    align=center    Color=Blue    Cursor=Hand </C>'
                       + '<C>ID=Name     Name="사업자명"  width=200   align=left</C>'
                       + '<C>ID=BusiNo   Name="사업자번호" width=100   align=center    mask="' + MASK_BIZNO + '" </C>';
TO-BE 
SetSheetAndDataSet("sheet", "dataSet01");

var cols = [
          {Type:"Status",  SaveName:"ISTATUS",  Hidden:true}
        , {Type:"Text",    SaveName:"DATA_KEY", Hidden:true}
        , {Type:"Seq",     SaveName:"CURROW",   Width:"30",   Align:"Center",  Edit:false}
        , {Type:"Text",    SaveName:"CODE",     Width:"70",   Align:"Center",  Edit:false,  FontColor:"BLUE",  Pointer:"HAND"}
        , {Type:"Text",    SaveName:"NAME",     Width:"200",  Align:"Left",    Edit:false}
        , {Type:"Text",    SaveName:"BUSINO",   Width:"100",  Align:"Center",  Format:MASK_BIZNO,  Edit:false}
    ];

var headers = [{Text: "상태|DATA_KEY|순번|고객코드|사업자명|사업자번호", Align:"Center"}];

InitIBSheet(sheet, cols, headers);

// 조회된 컬럼에 마우스 커서(손가락 형태) 이벤트 
sheet.SetDataLinkMouse("CUSTCOCODE", true);

// 마지막 컬럼 넓이 고정
sheet.SetExtendLastCol(0);
