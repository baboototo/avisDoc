Tab 초기화 설정 

/*
 * IBSheet Tab Tilte
 */
function initWindowTab(){
    // Tab Object, Tab Item Title Array Object
    initComTab(sheetTab, ["상담", "예약", "계약"]);
}

/*
 * IBSheet Tab Change Event
 */
function sheetTab_OnChange(curidx, oldidx){
	
    sheet.Reset(); // 탭이 변경될 때 sheet 내용 초기화
	
    if(curidx == 0){
        // 상담
        initFormat01();
    }else if(curidx == 1){
        // 예약
        initFormat02();
    }else if(curidx == 2){
        // 계약
        initFormat03();
    }
}
※ 참조
initWindowTab()
-   initComTab(Tab 명, [Tab 제목 배열])

sheetTab_OnChange()
-   Tab 선택 후 Sheet 정보 변경
-   Tab명_OnChange(현재 선택된 Tab Index, 이전에 선택된 Tab Index)
