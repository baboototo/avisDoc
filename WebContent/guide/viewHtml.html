<!-- 버튼 박스에서 좌우로 버튼을 위치해야 할 경우 사용 -->
<div class="tb_basic_btn">
    <table width="100%" cellspacing="0" cellpadding="0" border="0">
        <tbody>
            <tr>
                <th>
                    <!-- 버튼을 왼쪽에 위치 시킬경우 th을 사용  -->
                    <span class="button black large">
                        <input type="button" onclick="" value="등록" />
                    </span>
                </th>
                <td>
                    <!-- 버튼을 오른쪽에 위치 시킬경우 td을 사용  -->
                    <span class="button black large">
                        <input type="button" onclick="" value="등록" />
                    </span>
                </td>
            </tr>
        </tbody>
    </table>
</div>
